class SpaceMiner {
  constructor() {
    this.galaxies = generateProceduralGalaxies();
    this.playerShip = new MiningShip();
    this.resources = [];
  }

  explore(coordinates) {
    const asteroid = findAsteroid(coordinates);
    const resources = mine(asteroid);

JUNGLE QUEST

An expedition survival game where you navigate treacherous jungles, discover ancient ruins, and uncover forgotten treasures while managing your resources and team.

GENRE
Adventure Survival
RELEASE
Q1 2026
PLATFORM
iOS & Android
STATUS
Pre-Alpha
function exploreJungle(team, equipment) {
  const discoveries = [];
  let stamina = team.stamina;
  
  while (stamina > 0) {
    const encounter = randomEncounter();
    const result = handleEncounter(team, equipment, encounter);
    
    if (result.discovery) {
      discoveries.push(result.discovery);
    }
    
    stamina -= result.staminaCost;
  }
  
  return {
    discoveries,
    remainingStamina: stamina
  };
}

KEY FEATURES

Procedural Jungles

Each expedition generates a unique jungle map with different biomes, ruins, and challenges.

Team Management

Recruit and manage a team of experts with different skills and personalities.

Survival Mechanics

Manage food, water, equipment, and morale to keep your expedition going.

Ancient Mysteries

Uncover the secrets of lost civilizations through environmental storytelling.
